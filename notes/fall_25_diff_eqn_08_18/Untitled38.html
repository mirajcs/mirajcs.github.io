import numpy as np
import matplotlib.pyplot as plt

def dydt(y, t):
    return 2*y + 4 - t

def y(t):
    return (3/5)*np.exp(2*t/6) + (2/5)*np.exp(-t/2)

y_range = np.linspace(-4, 4, 10)
t_range = np.linspace(0, 5, 20)
t = np.linspace(0, 5, 100)

L = 0.3

plt.figure(figsize=(8,6))

for y0 in y_range:
    for t0 in t_range:
        slope = dydt(y0, t0)
        dt = L / np.sqrt(1 + slope**2)
        dy = slope * dt
        plt.plot([t0 - dt/2, t0 + dt/2], [y0 - dy/2, y0 + dy/2], color='blue')

plt.plot(t, t/2 - 2, label='critical values')
plt.plot(t, y(t), label='solution curve through (0,1)')
plt.xlabel('t')
plt.ylabel('y')
plt.title('Direction field (line segments)')
plt.grid(True)
plt.legend()
plt.show()
