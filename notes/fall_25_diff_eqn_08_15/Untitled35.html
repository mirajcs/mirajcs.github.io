import numpy as np
import matplotlib.pyplot as plt

def dvdt(v):
    return 9.8 - 0.2*v

v_consts = [40, 45, 50,55, 60, 49]  # all three values
dt = 2
t_vals = np.linspace(0, 50, 15)

plt.figure()

for v_const in v_consts:
    slope = dvdt(v_const)
    dv = slope * dt
    plt.quiver(t_vals, np.full_like(t_vals, v_const), dt, dv, angles='xy', scale_units='xy', scale=1, label=f'v_const={v_const}')
    print(f"v_const={v_const}, slope={slope}")

plt.grid(True)
plt.xlabel('t')
plt.ylabel('v')
plt.show()
